89 void MutablePriorityQueue::heapifyUp(unsigned i) {
90  auto x = H[i];
91  while (i > 1 && *x < *H[parent(i)]) {
92  set(i, H[parent(i)]);
93  i = parent(i);
94  }

112 void MutablePriorityQueue::heapifyDown(unsigned i) {
113  auto x = H[i];
114  while (true) {
115  unsigned k = leftChild(i);
116  if (k >= H.size())
117  break;
118  if (k+1 < H.size() && *H[k+1] < *H[k])
119  k++; // right child of i
120  if ( ! (*H[k] < *x) )
121  break;
122  set(i, H[k]);
123  i = k;
124  }
125  set(i, x);
126 }
